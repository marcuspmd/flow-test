    Interface AssertionChecks

    Regras de validação estendidas para um campo específico

    Versão aprimorada das regras de assertion incluindo validações de tipo, existência, comprimento e outras verificações avançadas.

    Example

    assert:
      body:
        user_id:
          type: number
          greater_than: 0
        email:
          exists: true
          regex: "^[\\w-\\.]+@([\\w-]+\\.)+[\\w-]{2,4}$"
        items:
          length:
            greater_than: 0
            less_than: 100
    interface AssertionChecks {
        equals?: any;
        contains?: any;

        exists?: boolean;
        type?: "string" | "number" | "boolean" | "object" | "array";
        length?: { equals?: number; greater_than?: number; less_than?: number };
