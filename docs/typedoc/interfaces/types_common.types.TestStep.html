    Interface TestStep

    Define uma única etapa no fluxo de teste

    Representa um step individual de um teste, contendo a requisição a ser executada, validações, capturas e cenários condicionais.

    Example

    steps:
      - name: "Login do usuário"
        request:
          method: POST
          url: /auth/login
          body:
            username: "{{test_user}}"
            password: "{{test_password}}"
        assert:
          status_code: 200
          body.token:
            not_equals: null
        capture:
          auth_token: "body.token"
        continue_on_failure: false
    interface TestStep {
        name: string;
        request: RequestDetails;
        assert?: Assertions;
        capture?: Record<string, string>;
        scenarios?: ConditionalScenario[];
        continue_on_failure?: boolean;
    }
